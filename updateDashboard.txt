        public ActionResult AdminTRR()
        {
            TRR obj = new TRR();
            using (var db = new RiskAppEntities())
            {
                var activeRecords = dbObj.TRRs.Where(e => e.IsActive).ToList();
                return View(activeRecords);
            }

        }
        public ActionResult AssignTRR(int id)
        {
            var project = dbObj.TRRs.Find(id);
            if (project != null)
            {
                project.IsActive = !project.IsActive; // Set IsActive to false
                dbObj.SaveChanges();
            }
            return RedirectToAction("AdminTRR");
        }

        public ActionResult SaveRemarks(int srNo, string remarks)
        {
            try
            {
                using (var db = new RiskAppEntities())
                {
                    var record = db.IBCPs.Find(srNo);
                    if (record != null)
                    {
                        record.Remarks = remarks; // Update remarks
                        db.SaveChanges();
                        return Json(new { success = true });
                    }
                }
            }
            catch (Exception)
            {
                // Log the exception
            }
            return Json(new { success = false });
        }
        [HttpPost]
        public ActionResult RejectRecord(int srNo)
        {
            using (var db = new RiskAppEntities())
            {
                var record = dbObj.IBCPs.FirstOrDefault(x => x.Sr_No == srNo);
                if (record != null)
                {
                    // Update the record status to 'Rejected'
                    record.Status = "Rejected";
                    db.Entry(record).State = EntityState.Modified;
                    db.SaveChanges();
                    return Json(new { success = true });
                }
            }
            return Json(new { success = false });
        }
